CREATE TABLE Roles
(
	role_id_pk SERIAL PRIMARY KEY,
	role_name VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE Sessions
(
	session_id_pk SERIAL PRIMARY KEY,
	date_creation DATE NOT NULL DEFAULT(CURRENT_DATE),
	package_fk INT NOT NULL,
	CONSTRAINT package_id_fk FOREIGN KEY(package_fk) REFERENCES Packages(package_id_pk) ON DELETE CASCADE
);

CREATE TABLE SessionPlayers
(
	id_sessionPlayer_pk SERIAL PRIMARY KEY,
	player_fk INT NOT NULL,
	session_fk INT NOT NULL,
	roles_fk INT NOT NULL,
	CONSTRAINT player_id_fk FOREIGN KEY(player_fk) REFERENCES Players(player_id_pk) ON DELETE CASCADE,
	CONSTRAINT session_id_fk FOREIGN KEY(session_fk) REFERENCES Sessions(session_id_pk) ON DELETE CASCADE,
	CONSTRAINT roles_id_fk FOREIGN KEY(roles_fk) REFERENCES Roles(role_id_pk) ON DELETE CASCADE
);
